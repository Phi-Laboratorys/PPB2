\begin{itemize}
    \item Startpunkt M1: $8,0762\,\text{mm}$
    \item Startpunkt M2: $42,2800\,\text{mm}$
    \item Graufilter: 1,6
    \item Bandpassfilter: orange
\end{itemize}
